set(source
 TestingPlugin.h
 TestingPlugin.cpp
 ../include/core.testing.plugin.h
)
set(moc
 TestingPlugin.h
)
set(ui
testingview.ui
)

source_group("" FILES ${source})

find_package(Qt4 COMPONENTS QtCore QtGui QtOpenGL qtmain REQUIRED)
include(${QT_USE_FILE})
include_directories(${CMAKE_CURRENT_BINARY_DIR})
QT4_WRAP_CPP(mocFiles ${moc})
QT4_WRAP_UI(uiFiles ${ui})

source_group("" FILES ${source})
source_group("autogenerated" FILES ${mocFiles} ${uiFiles})
source_group("ui" FILES ${ui})

add_library(core.testing.plugin ${source} ${mocFiles} ${uiFiles} CMakeLists.txt)


target_link_libraries(core.testing.plugin
  application.qt
  application.qt.objectview
  userinterface.qt.controls
  core.testing
  )
